```ts
import { actionFactory } from '@zedux/react'
```

A factory for creating [action factories](../types/ActionFactory) (yes, a "factory factory").

## Example

```ts live resultVar=output
const test = actionFactory('test')
const output = test({ some: 'data' })
```

## Signature

<Tabs>
  {tab1(`actionFactory = (type) => factory`)}
  {tab2(`declare const actionFactory: <
  Payload = undefined,
  Type extends string = string
>(
  actionType: Type
) => Payload extends undefined
  ? ActionFactory<undefined, Type>
  : ActionFactory<Payload, Type>`)}
</Tabs>

{<Legend>
  <Item name="type">
    <p>Required. A string.</p>
    <p>
      Will be set as the ActionFactory's <code>.type</code> property. Will also
      be the <code>type</code> property of all action objects created by the
      returned factory.
    </p>
  </Item>
  <Item name="Returns">
    <p>
      An <Link to="../types/ActionFactory">ActionFactory</Link> function. This
      factory accepts action payloads and returns valid action objects ready to
      be dispatched to{' '}
      <Link to="../classes/Store#dispatch">
        <code>store.dispatch()</code>
      </Link>
      .
    </p>
    <p>
      The action objects returned by this factory will have their{' '}
      <code>.type</code> and <code>.payload</code> properties set like so:
    </p>
    <Ts>{`const action = actionFactory('type')('payload')
action // { type: 'type', payload: 'payload' }`}</Ts>
  </Item>
</Legend>}
